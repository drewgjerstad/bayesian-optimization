% Theme Colors
\definecolor{Black}{RGB}{0, 0, 0}
\definecolor{White}{RGB}{255, 255, 255}
\definecolor{NavyBlue}{RGB}{1, 73, 124}

% Set Color Palette
\setbeamercolor{background canvas}{bg=White}
\setbeamercolor{frametitle}{bg=NavyBlue, fg=White}
\setbeamercolor{title}{bg=NavyBlue, fg=White}
\setbeamercolor{subtitle}{bg=NavyBlue, fg=White}
\setbeamercolor{author}{fg=Black}
\setbeamercolor{institute}{fg=Black}
\setbeamercolor{date}{fg=Black}
\setbeamercolor{item}{fg=Black}
\setbeamercolor{block title}{bg=White, fg=NavyBlue}
\setbeamercolor{block body}{bg=White, fg=NavyBlue}
\setbeamercolor{footline}{bg=NavyBlue, fg=White}
\setbeamercolor{author in footer}{bg=NavyBlue, fg=White}
\setbeamercolor{section in footer}{bg=NavyBlue, fg=White}
\setbeamercolor{page number in footer}{bg=NavyBlue, fg=White}
\setbeamercolor{section in toc}{fg=NavyBlue}
\setbeamercolor{subsection in toc}{fg=Black}

% Set Font Theme
%\usefonttheme{serif}

% Set Caption Name Color and Numbering
\setbeamertemplate{caption}[numbered]
\setbeamercolor{caption name}{fg=SlateGray}

% Customize Footer
\setbeamertemplate{footline}{%
    \leavevmode%
    \begin{beamercolorbox}[wd=\paperwidth, ht=2.5ex, dp=1ex, center]{footline}
        \begin{tabular*}{\paperwidth}[pos]{@{\extracolsep{\fill}} l c r}
            \hspace{1ex}\usebeamercolor[fg]{author in footer}\insertshortauthor &
            \usebeamercolor[bg]{section in footer}\usebeamercolor[fg]{section in footer}\insertsection &
            \usebeamercolor[fg]{page number in footer}\insertframenumber{} / \inserttotalframenumber\hspace{1ex}%
        \end{tabular*}
    \end{beamercolorbox}
}

% Remove navigation symbols
\setbeamertemplate{navigation symbols}{}
